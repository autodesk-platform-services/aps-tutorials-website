router.get('/api/issues/subtypes', async function(req, res, next){
    try {
        const subTypes = await getIssueSubtypes(req.query.projectId,req.internalOAuthToken.access_token);
        res.json(subTypes);
    } catch (err) {
        next(err);
    }
});
